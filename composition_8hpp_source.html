126 class all_result_submitter : public std::enable_shared_from_this<
127  all_result_submitter<T, Submissions, Args...>>,
128  public util::non_movable {

135  template <std::size_t From, std::size_t To, typename... PartialArgs>
136  void resolve(traits::size_constant<From> from, traits::size_constant<To>,
137  PartialArgs&&... args) {
138 
139  static_assert(sizeof...(args) == (To - From),
140  "Submission called with the wrong amount of arguments!");
141 
142  // Assign the values from the result to it's correct positions of the
143  // tuple. Maybe think about the thread safety again...:
144  // http://stackoverflow.com/questions/40845699
145  assign(from, result_, std::forward<PartialArgs>(args)...);
146 
147  // Complete the current result
148  complete_one();
149  }
150 
151  template <std::size_t From, std::size_t To>
152  void resolve(traits::size_constant<From>, traits::size_constant<To>,
153  types::dispatch_error_tag tag, types::error_type error) {
154 
155  // We never complete the composition, but we forward the first error
156  // which was raised.
157  std::call_once(flag_, std::move(callback_), tag, std::move(error));
158  }
159 
160  // Invokes the callback with the cached result
161  void invoke() {
162  assert((left_ == 0U) && "Expected that the submitter is finished!");
163  std::atomic_thread_fence(std::memory_order_acquire);
164  traits::unpack(std::move(result_), [&](auto&&... args) {
165  std::call_once(flag_, std::move(callback_),
166  std::forward<decltype(args)>(args)...);
167  });
168  }
169  // Completes one result
170  void complete_one() {
171  assert((left_ > 0U) && "Expected that the submitter isn't finished!");
172 
173  auto const current = --left_;
174  if (!current) {
175  invoke();
176  }
177  }

302 template <typename Strategy, typename LData, typename LAnnotation,
303  typename RData, typename RAnnotation>
304 auto connect(Strategy strategy, continuable_base<LData, LAnnotation>&& left,
305  continuable_base<RData, RAnnotation>&& right) {
306 
307  auto ownership_ =
308  base::attorney::ownership_of(left) | base::attorney::ownership_of(right);
309 
310  left.freeze();
311  right.freeze();
312 
313  // Make the new data which consists of a tuple containing
314  // all connected continuables.

316  normalize(strategy, std::move(right)));

319  // the current strategy as annotation.
320  return base::attorney::create(std::move(data), strategy, ownership_);

342 template <typename Data>
343 auto finalize_composition(
344  continuable_base<Data, strategy_all_tag>&& continuation) {
345 
346  auto ownership_ = base::attorney::ownership_of(continuation);
347 
348  auto composition = base::attorney::consume_data(std::move(continuation));
349 
350  // Merge all signature hints together
351  constexpr auto const signature = traits::unpack(composition, entry_merger{});
352 
353  return base::attorney::create(
354  [ signature,
355  composition = std::move(composition) ](auto&& callback) mutable {
356  // We mark the current 2-dimensional position through a pair:
357  // std::pair<size_constant<?>, size_constant<?>>
358  // ~~~~~~~~~~~~~~~~ ~~~~~~~~~~~~~~~~
359  // Continuation pos Result pos
360  constexpr auto const begin = std::make_pair(traits::size_constant_of<0>(),
361  traits::size_constant_of<0>());
362  constexpr auto const pack = traits::identify<decltype(composition)>{};
363  constexpr auto const end = traits::pack_size_of(pack);
364  auto const condition = [=](auto pos) { return pos.first < end; };
365 
366  // Create the result submitter which caches all results and invokes
367  // the final callback upon completion.
368  auto submitter = make_all_result_submitter(
369  std::forward<decltype(callback)>(callback), end, signature);
370 
371  // Invoke every continuation with it's callback of the submitter
372  traits::static_while(begin, condition, [&](auto current) mutable {
373  auto entry =
374  std::move(std::get<decltype(current.first)::value>(composition));
375 
376  // This is the length of the arguments of the current continuable
377  constexpr auto const arg_size =
378  traits::pack_size_of(hints::hint_of(traits::identity_of(entry)));
379 
380  // The next position in the result tuple
381  constexpr auto const next = current.second + arg_size;
382 
383  // Invoke the continuation with the associated submission callback
384  base::attorney::invoke_continuation(
385  std::move(entry),
386  submitter->create_callback(current.second, next));
387 
388  return std::make_pair(current.first + traits::size_constant_of<1>(),
389  next);
390  });
391  },
392  signature, std::move(ownership_));
393 }

454  continuable_base<Data, strategy_any_tag>&& continuation) {
