60 template <typename Continuable>
61 class awaitable {
62  using trait_t = util::expected_result_trait_t<Continuable>;
63 
65  Continuable continuable_;
68  typename trait_t::expected_type result_;
69 
70 public:
71  explicit constexpr awaitable(Continuable&& continuable)
72  : continuable_(std::move(continuable)) {
73  }
74 
77  bool await_ready() const noexcept {
78  return false;
79  }
80 
82  // TODO Convert this to an r-value function once possible
83  void await_suspend(coroutine_handle<> h) {
84  // Forward every result to the current awaitable
85  std::move(continuable_)
86  .next([h, this](auto&&... args) mutable {
87  resolve(std::forward<decltype(args)>(args)...);
88  h.resume();
89  })
90  .done();
91  }
92 
94  auto await_resume() noexcept(false) {
95  if (result_) {
96  // When the result was resolved return it
97  return trait_t::unwrap(std::move(result_));
98  }
99 
100 #if defined(CONTINUABLE_WITH_EXCEPTIONS)
101  std::rethrow_exception(result_.get_exception());
102 #else // CONTINUABLE_WITH_EXCEPTIONS
103  // Returning error types in await isn't supported as of now
104  util::trap();
105 #endif // CONTINUABLE_WITH_EXCEPTIONS
106  }
107 
108 private:
110  template <typename... Args>
111  void resolve(Args&&... args) {
112  result_.set_value(trait_t::wrap(std::forward<Args>(args)...));
113  }
114 
116  void resolve(types::dispatch_error_tag, types::error_type error) {
117  result_.set_exception(std::move(error));
118  }
119 };
